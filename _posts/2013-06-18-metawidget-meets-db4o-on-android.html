    3.1 Application</h4>
    
Bevor wir mit dem Speichern beginnen können, müssen wir uns eine db4o Datenbank anlegen. Dazu habe ich für unsere App eine neue Klasse "PersonApplication.java" erstellt und diese von Application erben lassen. Das hat den Vorteil, dass wir von unseren beiden Activities mit getApplication() darauf zugreifen können.
    

    Den Zugriff auf den Datenbank-File realisieren wir mit Db4oEmbedded.openfile(...). Sollte keine Datei vorhanden sein, wir automatisch eine neue Angelegt. Über den zurückgelieferten ObjectContainer können wir die Befehle für das Speichern/Löschen/Bearbeiten absetzen.
    

    public class PersonApplication extends Application {
    

    	private final String DATABASE_FILE = "person.dbo";
    
	private ObjectContainer oContainer;
    

    	public PersonApplication(){
    
        super();
    
	}
    

    	private ObjectContainer initDatabase(){
    
		try{
    
			if(oContainer == null || oContainer.ext().isClosed()){
    
				oContainer = Db4oEmbedded.openFile(dbConfig(), db4oDBFullPath(this));
    
				return oContainer;
    
			}
    
		} catch (Exception e){
    
			e.printStackTrace();
    
		}
    
		return oContainer;
    
	}
    

    	private EmbeddedConfiguration dbConfig() {
    
		EmbeddedConfiguration config = Db4oEmbedded.newConfiguration();
    
		return config;
    
	}
    

    	private String db4oDBFullPath(Context context) {
    
		return context.getDir("data",0) + "/" + DATABASE_FILE;
    
	}
    

    	public ObjectContainer getDatabase(){
    
		return initDatabase();
    
	}
    
}</pre>
    
Wichtig ist hierbei, dass wir unsere AndroidManifest.xml um eine Zeile erweitern und zwar muss bei  noch diese Zeile eingefügt werden.

    3.2 Speichern</h4>
    
Nachdem wir jetzt eine Möglichkeit haben unsere Datenbank zu erreichen, können wir unsere PersonActivity weiter anpassen. Dazu passen wir die Klasse an, dass diese folgendermaßen aussieht.
    

    public class PersonActivity extends Activity {
    

    	private AndroidMetawidget aMetawidget;
    
	private Button saveButton;
    
	private ObjectContainer db;
    

    	@Override
    
	protected void onCreate(Bundle savedInstanceState) {
    
		super.onCreate(savedInstanceState);
    
		setContentView(R.layout.person_activity_layout);
    
		if(db == null){
    
			 db = ((PersonApplication) getApplication()).getDatabase();
    
		}
    
		person = new Person();
    

    		aMetawidget = (AndroidMetawidget) findViewById(R.id.metawidget);
    
		aMetawidget.setToInspect(person);
    

    		saveButton = (Button) findViewById(R.id.saveButton);
    
		saveButton.setOnClickListener(new OnClickListener() {
    
			@Override
    
			public void onClick(View v) {
    
				if(validateFields()){
    
					person.setFirstName(aMetawidget.getValue("firstName").toString());
    
					person.setLastName(aMetawidget.getValue("lastName").toString());
    
					person.setAge(Integer.parseInt(aMetawidget.getValue("age").toString()));
    
					person.setStreetName(aMetawidget.getValue("streetName").toString());
    
					person.setPostalCode(Integer.parseInt(aMetawidget.getValue("postalCode").toString()));
    
					person.setCity(aMetawidget.getValue("city").toString());
    
					db.store(person);
    
                    db.commit();
    
					finish();
    
				}
    
			}
    
		});
    
	}
    

    	private boolean validateFields(){
    
		if(TextUtils.isEmpty(aMetawidget.getValue("firstName").toString())
    
		|| TextUtils.isEmpty(aMetawidget.getValue("lastName").toString())
    
		|| TextUtils.isEmpty(aMetawidget.getValue("age").toString())
    
		|| TextUtils.isEmpty(aMetawidget.getValue("streetName").toString())
    
		|| TextUtils.isEmpty(aMetawidget.getValue("postalCode").toString())
    
		|| TextUtils.isEmpty(aMetawidget.getValue("city").toString())){
    
			showToast();
    
			return false;
    
		} else {
    
			return true;
    
		}
    

    	}
    

    	private void showToast() {
    
		Toast.makeText(this, "Eingabe überprüfen!", Toast.LENGTH_SHORT).show();

    public class MainActivity extends ListActivity {
    

    	private ObjectContainer db;
    
	private Person[] values;
    

    	@Override
    
	protected void onCreate(Bundle savedInstanceState) {
    
		super.onCreate(savedInstanceState);
    
		setContentView(R.layout.activity_main);
    
		if(db == null){
    
			 db = ((PersonApplication) getApplication()).getDatabase();
    
		}
    
		setListContent();
    
	}
    

    	private void setListContent() {
    
		ObjectSet result = db.queryByExample(Person.class);
    
		listResult(result);
    
		ArrayAdapter adapter = new ArrayAdapter(this, android.R.layout.simple_list_item_1, values);
    
		setListAdapter(adapter);
    
	}
    

    	private void listResult(ObjectSet result) {
    
		if(!result.isEmpty()){
    
			values = new Person[result.size()];
    
			for(int i = 0; i < result.size(); i++){
    
				values[i] = (result.get(i));
    
			}
    
		} else {
    
			values = new Person[]{new Person()};
    
		}
    
	}
    

    	@Override
    
	protected void onResume() {
    
		setListContent();
    
		super.onResume();
    
	}
    

    	@Override
    
	public boolean onCreateOptionsMenu(Menu menu) {
    
		getMenuInflater().inflate(R.menu.main, menu);
    
		return true;
    
	}
    

    	@Override
    
	public boolean onOptionsItemSelected(MenuItem item) {
    
		switch (item.getItemId()) {
    
		case R.id.addPerson:
    
			Intent intent = new Intent(this, PersonActivity.class);
    
			startActivity(intent);
    
			return true;
    
		default:
    
			return super.onOptionsItemSelected(item);
    
		}
    
	}
